## 큐

큐: 먼저 들어온 데이터가 먼저 나가는 자료구조<br/>선입선출(FIFO : First - In First - Out)

삽입: 후단(rear)<br/>삭제: 전단(front)

## 선형큐(1차원 배열)

```c
#include <stdio.h>
#include <stdlib.h>
#define MAX_QUEUE_SIZE 5

typedef int element;
typedef struct {
	int front;
	int rear;
	element data[MAX_QUEUE_SIZE];
} QueueType;

// 오류 함수
void error(const char* message) { 
	fprintf(stderr, "%s\n", message);
	exit(1);
}

// 초기화
void init_queue(QueueType* q) { 
	q->rear = -1;
	q->front = -1;
}

// print
void queue_print(QueueType* q) {
	for (int i = 0; i < MAX_QUEUE_SIZE; i++) {
		if (i <= q->front || i > q->rear) printf("   | ");
		else printf("%d | ", q->data[i]);
	}printf("\n");
}

// 포화상태 검출
int is_full(QueueType* q) {
	if (q->rear == MAX_QUEUE_SIZE - 1) return 1;
	else return 0;
}

// 공백상태 검출
int is_empty(QueueType* q) {
	if (q->front == q->rear) return 1;
	else return 0;
}

// ++rear에 삽입
void enqueue(QueueType* q, int item) {
	if (is_full(q)) {
		error("큐가 포화상태입니다.");
		return;
	}
	q->data[++(q->rear)] = item; 
}

// 삭제: ++front 반환(front는 맨 앞 가리키고 있음)
int dequeue(QueueType* q) {
	if (is_empty(q)) {
		error("큐가 공백상태입니다.");
		return -1;
	}
	int item = q->data[++(q->front)];
	return item;
}

int main() {
	QueueType q;

	init_queue(&q);//초기화
	enqueue(&q, 10); queue_print(&q); // 10 삽입
	enqueue(&q, 20); queue_print(&q); // 20 삽입
	enqueue(&q, 30); queue_print(&q); // 30 삽입

	dequeue(&q); queue_print(&q); // 10 삭제
	dequeue(&q); queue_print(&q); // 20 삭제
	dequeue(&q); queue_print(&q); // 30 삭제
	return 0;
}
```
```
10 |    |    |    |    |
10 | 20 |    |    |    |
10 | 20 | 30 |    |    |
   | 20 | 30 |    |    |
   |    | 30 |    |    |
   |    |    |    |    |
```

## 선형큐(링크드 리스트)

```c
#include <stdio.h>
#include <stdlib.h>

typedef int element;
typedef struct QueueNode { // 큐 노드
	element data;
	struct QueueNode* link;
} QueueType;

typedef struct {
	QueueNode* front, * rear;
}LinkedQueueType;

// 오류 함수
void error(const char* message) { 
	fprintf(stderr, "%s\n", message);
	exit(1);
}

// 초기화
void init_queue(LinkedQueueType* q) { 
	q->rear = NULL;
	q->front = NULL;
}

// print
void queue_print(LinkedQueueType* q) {
	printf("   | ");
	for(QueueNode*p= q->front; p!=NULL; p=p->link){
		printf("%d | ", p->data);
	}
	printf("\n");
}

// 공백상태 검출
int is_empty(LinkedQueueType* q) {
	return (q->front == NULL); // boolean
}

// 삽입
void enqueue(LinkedQueueType* q, element data) {
	QueueNode* temp = (QueueNode*)malloc(sizeof(QueueNode)); // 동적메모리 할당
	if (temp == NULL) error("메모리 할당 에러");
	
	temp->data = data; // temp node에 데이터 저장
	temp->link = NULL; // 링크필드 NULL

	if (is_empty(q)) {		// (q->front == NULL)이면 (공백)
		q->front = temp;	// front, rear 둘 다 temp에 연결
		q->rear = temp;
	}
	else {					  // 순서가 중요!
		q->rear->link = temp; // rear link에 연결해주고
		q->rear = temp;		  // rear(후단) 위치 변경
	}
}

// 삭제
int dequeue(LinkedQueueType* q) {
	QueueNode* temp = q->front;
	element data;

	if (is_empty(q)) error("큐가 공백상태입니다.");

	data = temp->data;
	q->front = q->front->link; // front 뒤에 front 달아주기

	if (q->front == NULL) q->rear = NULL; // 큐 공백
	
	free(temp); // 동적 메모리 해제
	return data; 
}

int main() {
	LinkedQueueType q;

	init_queue(&q);//초기화
	enqueue(&q, 10); queue_print(&q); // 10 삽입
	enqueue(&q, 20); queue_print(&q); // 20 삽입
	enqueue(&q, 30); queue_print(&q); // 30 삽입

	dequeue(&q); queue_print(&q); // 10 삭제
	dequeue(&q); queue_print(&q); // 20 삭제
	dequeue(&q); queue_print(&q); // 30 삭제
	return 0;
}
```
```
   | 10 |
   | 10 | 20 |
   | 10 | 20 | 30 |
   | 20 | 30 |
   | 30 |
   |
```

## 원형큐
초기: front, rear--> [0]<br/>
삽입: rear--> [1]<br/>
삭제: front--> [1]

공복: front == rear<br/>
포화: front % M == (rear + 1) % M (front가 rear보다 하나 앞에 있으면 포화상태이다.)
##### * 원형큐에서는 포화/공백 상태 구별을 위해 한 자리는 항상 비워둔다.
###### --> count 변수를 사용하면 비워두지 않아도 된다.

```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#define MAX_QUEUE_SIZE 5

typedef int element;
typedef struct {
	element data[MAX_QUEUE_SIZE];
	int front, rear;
}QueueType;

// 오류 함수
void error(const char* message) {
	fprintf(stderr, "%s\n", message);
	exit(1);
}

// 초기화
void init_queue(QueueType* q) {
	q->front = q->rear = 0;
}// 초기: front, rear--> [0]

// 포화 상태 검출
int is_full(QueueType* q) {
	return ((q->rear + 1) % MAX_QUEUE_SIZE == q->front % MAX_QUEUE_SIZE);
}

// 공백상태 검출
int is_empty(QueueType* q) {
	return (q->front == q->rear);
}

// print
void queue_print(QueueType* q) {
	printf("QUEUE(front=%d rear=%d) = ", q->front, q->rear);
	if (!is_empty(q)) {   // 들어가 있으면
		int i = q->front;
		do {
			i = (i + 1) % (MAX_QUEUE_SIZE);
			printf("%d | ", q->data[i]);
			if (i == q->rear) break;
		} while (i != q->front);
	}printf("\n");
}

// 삽입
void enqueue(QueueType* q, int item) {
	if (is_full(q)) error("큐가 포화상태입니다.");
	q->rear = (q->rear + 1) % MAX_QUEUE_SIZE;
	q->data[q->rear] = item;
}

// 삭제
element dequeue(QueueType* q) {
	if (is_empty(q)) error("큐가 공백상태입니다.");
	q->front = (q->front + 1) % MAX_QUEUE_SIZE;
	return q->data[q->front];
}

int main() {
	QueueType q;
	int element;

	init_queue(&q);//초기화
	while (!is_full(&q)) {
		printf("정수를 입력하시오: ");
		scanf("%d", &element);

		enqueue(&q, element);
		queue_print(&q);
	}
	printf("\n");
	while (!is_empty(&q)) {
		element = dequeue(&q);
		printf("꺼내진 정수: %d \n", element);
		queue_print(&q);
	}
	return 0;
}
```
```
정수를 입력하시오: 1
QUEUE(front=0 rear=1) = 1 |
정수를 입력하시오: 2
QUEUE(front=0 rear=2) = 1 | 2 |
정수를 입력하시오: 3
QUEUE(front=0 rear=3) = 1 | 2 | 3 |
정수를 입력하시오: 4
QUEUE(front=0 rear=4) = 1 | 2 | 3 | 4 |

꺼내진 정수: 1
QUEUE(front=1 rear=4) = 2 | 3 | 4 |
꺼내진 정수: 2
QUEUE(front=2 rear=4) = 3 | 4 |
꺼내진 정수: 3
QUEUE(front=3 rear=4) = 4 |
꺼내진 정수: 4
QUEUE(front=4 rear=4) =
```

## 덱
**덱(deque)**: **double-ended queue**의 줄임말로서<br/>
큐의 전단과 후단에서 모두 삽입과 삭제가 가능한 큐<br/>
##### * get_#(): 공백이 아닐 때 #이 가리키고 있는 항목 반환<br/>
##### * %: 삽입 시 (M-1)부터 front는 작아지고 rear는 커지면서 회전
##### * 원형큐에서는 포화/공백 상태 구별을 위해 한 자리는 항상 비워둔다.
```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#define MAX_QUEUE_SIZE 5 // 0~4

typedef int element;
typedef struct {
	element data[MAX_QUEUE_SIZE];
	int front, rear;
}DequeType;

// 오류 함수
void error(const char* message) {
	fprintf(stderr, "%s\n", message);
	exit(1);
}

// 초기화
void init_deque(DequeType* q) {
	q->front = q->rear = 0;
}// 초기: front, rear--> [0]

// 포화 상태 검출
int is_full(DequeType* q) {
	return ((q->rear + 1) % MAX_QUEUE_SIZE == q->front % MAX_QUEUE_SIZE);
}

// 공백상태 검출
int is_empty(DequeType* q) {
	return (q->front == q->rear);
}

// print
void deque_print(DequeType* q) {
	printf("DEQUE(front=%d rear=%d) = ", q->front, q->rear);
	if (!is_empty(q)) {   // 들어가 있으면
		int i = q->front;
		do {
			i = (i + 1) % (MAX_QUEUE_SIZE);
			printf("%d | ", q->data[i]);
			if (i == q->rear) break;
		} while (i != q->front);
	}printf("\n");
}

// 삽입(front)
void add_front (DequeType* q, element val) {
	if (is_full(q)) error("덱이 포화상태입니다.");
	q->data[q->front] = val;
	q->front = (q->front - 1 + MAX_QUEUE_SIZE) % MAX_QUEUE_SIZE;
}
// 삽입(rear)
void add_rear(DequeType* q, element item) {
	if (is_full(q)) error("덱이 포화상태입니다.");
	q->rear = (q->rear +1) % MAX_QUEUE_SIZE;
	q->data[q->rear] = item;
}

// 삭제(front)
element delete_front(DequeType* q) {
	if (is_empty(q)) error("덱이 공백상태입니다.");
	q->front = (q->front + 1) % MAX_QUEUE_SIZE;
	return q->data[q->front];
}
// 삭제(rear)
element delete_rear(DequeType* q) {
	int prev = q->rear;
	if (is_empty(q)) error("덱이 공백상태입니다.");
	q->rear = (q->rear - 1 + MAX_QUEUE_SIZE) % MAX_QUEUE_SIZE;
	return q->data[prev];
}
// front가 가리키고 있는 요소 반환
element get_front(DequeType* q) {
	if (is_empty(q)) error("덱이 공백상태입니다.");
	return q->data[(q->front + 1) % MAX_QUEUE_SIZE];
}// rear가 가리키고 있는 요소 반환
element get_rear(DequeType* q) {
	if (is_empty(q)) error("덱이 공백상태입니다.");
	return q->data[q->rear];
}

int main() {
	DequeType q;
	int element;

	init_deque(&q); //초기화
	while (!is_full(&q)) {
		printf("정수를 입력하시오: ");
		scanf("%d", &element);

		add_front(&q, element);
		deque_print(&q);
	}
	printf("\n");
	while (!is_empty(&q)) {
		element = delete_rear(&q);
		printf("꺼내진 정수: %d \n", element);
		deque_print(&q);
	}
	return 0;
}
```
